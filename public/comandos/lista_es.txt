<script type="module">
    import firebaseConfig from './config.js';
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
    initializeApp(firebaseConfig);

    const translations = {
        pt: { file: 'lista_cmd_bash.txt', subtitle: "Toque no comando para copiar", placeholder: "Buscar comando...", modalTitle: "Sobre", modalDesc: "Buscador para Termux.", devs: "Devs:", close: "FECHAR", copied: "COPIADO!", share: "Olha esse buscador Bash!" },
        en: { file: 'lista_en.txt', subtitle: "Tap command to copy", placeholder: "Search command...", modalTitle: "About", modalDesc: "Searcher for Termux.", devs: "Devs:", close: "CLOSE", copied: "COPIED!", share: "Check this Bash searcher!" },
        es: { file: 'lista_es.txt', subtitle: "Toca el comando para copiar", placeholder: "Buscar comando...", modalTitle: "Sobre", modalDesc: "Buscador para Termux.", devs: "Devs:", close: "CERRAR", copied: "¡COPIADO!", share: "¡Mira este buscador!" }
    };

    let currentLang = 'pt';
    let baseDados = [];

    async function carregar() {
        const t = translations[currentLang];
        document.getElementById('ui-subtitle').innerText = t.subtitle;
        document.getElementById('search').placeholder = t.placeholder;
        document.getElementById('ui-modal-title').innerText = t.modalTitle;
        document.getElementById('ui-modal-desc').innerText = t.modalDesc;
        document.getElementById('ui-devs').innerText = t.devs;
        document.getElementById('ui-close').innerText = t.close;
        document.getElementById('langBtn').innerText = currentLang.toUpperCase();

        const res = await fetch('comandos/' + t.file);
        const texto = await res.text();
        baseDados = texto.split('\n').filter(l => l.includes(' - ')).map(l => {
            const [n, d] = l.split(' - '); return { nome: n.trim(), desc: d.trim() };
        });
        mostrar(baseDados);
    }

    document.getElementById('langBtn').onclick = () => {
        currentLang = currentLang === 'pt' ? 'en' : (currentLang === 'en' ? 'es' : 'pt');
        carregar();
    };

    window.toggleModal = (s) => document.getElementById('aboutModal').style.display = s ? 'flex' : 'none';
    window.copiarComando = (el, txt) => {
        navigator.clipboard.writeText(txt).then(() => {
            el.classList.add('copied');
            setTimeout(() => el.classList.remove('copied'), 1500);
        });
    };

    function mostrar(itens) {
        document.getElementById('lista').innerHTML = itens.map(i => `
            <li onclick="copiarComando(this, '${i.nome}')">
                <span class="copy-badge">${translations[currentLang].copied}</span>
                <span class="cmd">$ ${i.nome}</span>
                <span class="desc">${i.desc}</span>
            </li>`).join('');
    }

    document.getElementById('search').oninput = (e) => {
        const val = e.target.value.toLowerCase();
        mostrar(baseDados.filter(i => i.nome.includes(val) || i.desc.includes(val)));
    };

    carregar();
    if ("serviceWorker" in navigator) navigator.serviceWorker.register("sw.js");
</script>
